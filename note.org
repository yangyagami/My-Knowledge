* qml 监听并且设置快捷键
[2025-06-11 三 15:57]
#+BEGIN_SRC qml
  Button {
      id: toggleWakeup
      anchors.right: parent.right
      anchors.verticalCenter: parent.verticalCenter
      text: "点击我或按 Ctrl+S"
      onClicked: {
  	toggleWakeup.text = "按钮被点击或 Ctrl+S 被按下";
      }

      Keys.onPressed: {
  	if (event.modifiers & Qt.ControlModifier && event.key === Qt.Key_S) {
  	    toggleWakeup.text = "你按下了 Ctrl+S";
  	    event.accepted = true;
  	}
      }
  }  // Button($toggleWakeup)
#+END_SRC
